<link rel="stylesheet" href="/glossaryCanopy/app/styles/custom.css" />

import {
  Label,
  Summary,
  Metadata,
  RequiredStatement,
  References,
} from "@canopy-iiif/app/ui/server";

import { useEffect } from "react";

export default function WorkLayout(props) {

  useEffect(() => {
    const secondary = document.querySelector(".canopy-work--secondary");
    const glossaryDiv = document.getElementById("glossary-section");

    if (!secondary || !glossaryDiv) return;

    const loadGlossary = async () => {
      try {
        const { glossaryTerms } = await import("/glossaryCanopy/app/js/glossary-terms.js");
        const {
          identifyGlossaryMatches,
          highlightGlossaryTermsString,
          renderGlossary,
          attachGlossaryTooltips
        } = await import("/glossaryCanopy/app/js/glossary-functions.js");

        const textToScan = secondary.textContent || "";
        const matches = identifyGlossaryMatches(textToScan, glossaryTerms);

        // Highlight the main text
        secondary.innerHTML = highlightGlossaryTermsString(textToScan, matches);

        // Render the glossary panel
        renderGlossary(matches, glossaryDiv);

        // Attach tooltip functionality
        attachGlossaryTooltips(secondary);

      } catch (e) {
        console.error("Glossary initialization failed:", e);
      }
    };

    loadGlossary();
  }, []);

  return (
    <>
      <Container variant="wide" className="canopy-work--layout">
        <div className="canopy-work--primary">
          <Viewer iiifContent={props.manifest.id} />
        </div>

        <div className="canopy-work--secondary">
          <header>
            <Label manifest={props.manifest} as="h1" />
            <Summary manifest={props.manifest} as="p" className="canopy-lead" />
          </header>

          {/* Metadata and references */}
          <div>
            <References />
            <Metadata manifest={props.manifest} />
            <RequiredStatement manifest={props.manifest} />
          </div>

          {/* Glossary panel */}
          <div id="glossary-section"></div>
        </div>
      </Container>

      <Container>
        <h2>Related Items</h2>
        <RelatedItems iiifContent={props.manifest.id} top={3} />
      </Container>
    </>
  );
}
